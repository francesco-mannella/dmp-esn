319 {

321  if (goal_system_==NULL)
322  {
323  // If there is no dynamical system for the delayed goal, the goal is
324  // simply the attractor state
325  spring_system_->set_attractor_state(attractor_state());
326  // with zero change
327  xd.GOAL.fill(0);
328  }
329  else
330  {
331  // Integrate goal system and get current goal state
332  goal_system_->set_attractor_state(attractor_state());

334  // The goal state is the attractor state of the spring-damper system
335  spring_system_->set_attractor_state(x.GOAL);
336 
337  }
338 
339  // Integrate spring damper system
340  // Forcing term is added to spring_state later
341  spring_system_->differentialEquation(x.SPRING, xd.SPRING);
342 
343 
344  // Non-linear forcing term
345  phase_system_->differentialEquation(x.PHASE, xd.PHASE);
346  gating_system_->differentialEquation(x.GATING, xd.GATING);
347 
348  MatrixXd phase_state(1,1);
349  phase_state = x.PHASE;
350  MatrixXd fa_output(1,dim_orig());
351  computeFunctionApproximatorOutput(phase_state, fa_output);
352 
353  int t0 = 0;
354  double gating = (x.GATING)[0];
355  VectorXd g_minus_y0 = (attractor_state()-initial_state()).transpose();
356  VectorXd forcing_term = gating*fa_output.row(t0); // todo .array()*g_minus_y0.array();
357 
358  // Add forcing term to the ZD component of the spring state
359  xd.SPRING_Z = xd.SPRING_Z + forcing_term/tau();

403 void Dmp::analyticalSolution(const Eigen::VectorXd& ts, Eigen::MatrixXd& xs, Eigen::MatrixXd& xds, Eigen::MatrixXd& forcing_terms, Eigen::MatrixXd& fa_outputs) const
404 {
405  int n_time_steps = ts.size();
406  assert(n_time_steps>0);
407 
408  // Usually, we expect xs and xds to be of size T X dim(), so we resize to that. However, if the
409  // input matrices were of size dim() X T, we return the matrices of that size by doing a
410  // transposeInPlace at the end. That way, the user can also request dim() X T sized matrices.
411  bool caller_expects_transposed = (xs.rows()==dim() && xs.cols()==n_time_steps);
412 
413  // INTEGRATE SYSTEMS ANALYTICALLY AS MUCH AS POSSIBLE
414 
415  // Integrate phase
416  MatrixXd xs_phase;
417  MatrixXd xds_phase;
418  phase_system_->analyticalSolution(ts,xs_phase,xds_phase);
419 
420  // Compute gating term
421  MatrixXd xs_gating;
422  MatrixXd xds_gating;
423  gating_system_->analyticalSolution(ts,xs_gating,xds_gating);
424 
425  // Compute the output of the function approximator
426  fa_outputs.resize(ts.size(),dim_orig());
427  fa_outputs.fill(0.0);
428  //if (isTrained())
429  computeFunctionApproximatorOutput(xs_phase, fa_outputs);
430 
431  MatrixXd xs_gating_rep = xs_gating.replicate(1,fa_outputs.cols());
432  MatrixXd g_minus_y0_rep = (attractor_state()-initial_state()).transpose().replicate(n_time_steps,1);
433  forcing_terms = fa_outputs.array()*xs_gating_rep.array(); // todo *g_minus_y0_rep.array();
434 
435  MatrixXd xs_goal, xds_goal;
436  // Get current delayed goal
437  if (goal_system_==NULL)
438  {
439  // If there is no dynamical system for the delayed goal, the goal is
440  // simply the attractor state
441  xs_goal = attractor_state().transpose().replicate(n_time_steps,1);
442  // with zero change
443  xds_goal = MatrixXd::Zero(n_time_steps,dim_orig());
444  }
445  else
446  {
447  // Integrate goal system and get current goal state
448  goal_system_->analyticalSolution(ts,xs_goal,xds_goal);
449  }
450 
451  xs.resize(n_time_steps,dim());
452  xds.resize(n_time_steps,dim());
453 
454  int T = n_time_steps;
455 
456  xs.GOALM(T) = xs_goal; xds.GOALM(T) = xds_goal;
457  xs.PHASEM(T) = xs_phase; xds.PHASEM(T) = xds_phase;
458  xs.GATINGM(T) = xs_gating; xds.GATINGM(T) = xds_gating;
459 
460 
461  // THE REST CANNOT BE DONE ANALYTICALLY
462 
463  // Reset the dynamical system, and get the first state
464  double damping = spring_system_->damping_coefficient();
465  SpringDamperSystem localspring_system_(tau(),initial_state(),attractor_state(),damping);
466 
467  // Set first attractor state
468  localspring_system_.set_attractor_state(xs_goal.row(0));
469 
470  // Start integrating spring damper system
471  int dim_spring = localspring_system_.dim();
472  VectorXd x_spring(dim_spring), xd_spring(dim_spring); // todo Why are these needed?
473  int t0 = 0;
474  localspring_system_.integrateStart(x_spring, xd_spring);
475  xs.row(t0).SPRING = x_spring;
476  xds.row(t0).SPRING = xd_spring;
477 
478  // Add forcing term to the acceleration of the spring state
479  xds.SPRINGM_Z(1) = xds.SPRINGM_Z(1) + forcing_terms.row(t0)/tau();
480 
481  for (int tt=1; tt<n_time_steps; tt++)
482  {
483  double dt = ts[tt]-ts[tt-1];
484 
485  // Euler integration
486  xs.row(tt).SPRING = xs.row(tt-1).SPRING + dt*xds.row(tt-1).SPRING;
487 
488  // Set the attractor state of the spring system
489  localspring_system_.set_attractor_state(xs.row(tt).GOAL);
490 
491  // Integrate spring damper system
492  localspring_system_.differentialEquation(xs.row(tt).SPRING, xd_spring);
493  xds.row(tt).SPRING = xd_spring;
494 
495  // If necessary add a perturbation. May be useful for some off-line tests.
496  RowVectorXd perturbation = RowVectorXd::Constant(dim_orig(),0.0);
497  if (analytical_solution_perturber_!=NULL)
498  for (int i_dim=0; i_dim<dim_orig(); i_dim++)
499  // Sample perturbation from a normal Gaussian distribution
500  perturbation(i_dim) = (*analytical_solution_perturber_)();
501 
502  // Add forcing term to the acceleration of the spring state
503  xds.row(tt).SPRING_Z = xds.row(tt).SPRING_Z + forcing_terms.row(tt)/tau() + perturbation;
504  // Compute y component from z
505  xds.row(tt).SPRING_Y = xs.row(tt).SPRING_Z/tau();
506 
507  }
508 
509  if (caller_expects_transposed)
510  {
511  xs.transposeInPlace();
512  xds.transposeInPlace();
513  }

662 void Dmp::getParameterVectorMask(const std::set<std::string> selected_values_labels, Eigen::VectorXi& selected_mask) const
663 {
664  assert(function_approximators_.size()>0);
665  for (int dd=0; dd<dim_orig(); dd++)
666  {
667  assert(function_approximators_[dd]!=NULL);
668  assert(function_approximators_[dd]->isTrained());
669  }
670 
671  selected_mask.resize(getParameterVectorAllSize());
672  selected_mask.fill(0);
673 
674  const int TMP_GOAL_NUMBER = -1;
675  int offset = 0;
676  VectorXi cur_mask;
677  for (int dd=0; dd<dim_orig(); dd++)
678  {
679  function_approximators_[dd]->getParameterVectorMask(selected_values_labels,cur_mask);
680 
681  // This makes sure that the indices for each function approximator are different
682  int mask_offset = selected_mask.maxCoeff();
683  for (int ii=0; ii<cur_mask.size(); ii++)
684  if (cur_mask[ii]!=0)
685  cur_mask[ii] += mask_offset;
686 
687  selected_mask.segment(offset,cur_mask.size()) = cur_mask;
688  offset += cur_mask.size();
689 
690  // Goal
691  if (selected_values_labels.find("goal")!=selected_values_labels.end())
692  selected_mask(offset) = TMP_GOAL_NUMBER;
693  offset++;
694 
695  }
696  assert(offset == getParameterVectorAllSize());
697 
698  // Replace TMP_GOAL_NUMBER with current max value
699  int goal_number = selected_mask.maxCoeff() + 1;
700  for (int ii=0; ii<selected_mask.size(); ii++)
701  if (selected_mask[ii]==TMP_GOAL_NUMBER)
702  selected_mask[ii] = goal_number;
703
