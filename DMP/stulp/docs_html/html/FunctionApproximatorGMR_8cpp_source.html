197 double FunctionApproximatorGMR::normalPDF(const VectorXd& mu, const MatrixXd& covar, const VectorXd& input)
198 {
199  MatrixXd covar_inverse = covar.inverse();
200  double output = exp(-0.5*(input-mu).transpose()*covar_inverse*(input-mu));
201  // For invertible matrices (which covar apparently was), det(A^-1) = 1/det(A)
202  // Hence the 1.0/covar_inverse.determinant() below
203  // ( (2*pi)^N*|\Sigma| )^(-1/2)
204  output *= pow(pow(2*M_PI,mu.size())/covar_inverse.determinant(),-0.5);
205  return output;
206 }

225 void FunctionApproximatorGMR::predict(const Eigen::MatrixXd& inputs, Eigen::MatrixXd& outputs, Eigen::MatrixXd& variances)
226 {
227  ENTERING_REAL_TIME_CRITICAL_CODE
228 
229  // The reason this function is not pretty and so long (which I usually try to avoid) is to
230  // avoid Eigen making dynamic memory allocations. This would cause trouble in real-time critical
231  // code. Therefore, I
232  // * use preallocated matrices (member variables) for intermediate results
233  // * use noalias() whenever needed (http://eigen.tuxfamily.org/dox/TopicLazyEvaluation.html)
234  // * try to avoid calling other functions (a bit tricky when using Eigen matrices as parameters)
235  // I hope the documentation makes up for the ugliness...
236 
237  if (!isTrained())
238  {
239  cerr << "WARNING: You may not call FunctionApproximatorGMR::predict if you have not trained yet. Doing nothing." << endl;
240  return;
241  }
242 
243  const ModelParametersGMR* gmm = static_cast<const ModelParametersGMR*>(getModelParameters());
244 
245  // Number of Gaussians must be at least one
246  assert(gmm->getNumberOfGaussians()>0);
247  // Dimensionality of input must be same as of the gmm inputs
248  assert(gmm->getExpectedInputDim()==inputs.cols());
249 
250  // Only compute the means if the outputs matrix is not empty
251  bool compute_means = false;
252  if (outputs.rows()>0)
253  {
254  outputs.resize(inputs.rows(),gmm->getExpectedOutputDim());
255  outputs.fill(0);
256  compute_means = true;
257  }
258 
259  // Only compute the variances if the variances matrix is not empty
260  bool compute_variance = false;
261  if (variances.rows()>0)
262  {
263  compute_variance = true;
264  variances.resize(inputs.rows(),gmm->getExpectedOutputDim());
265  variances.fill(0);
266  }
267 
268  // For each input, compute the output
269  for (int i_input=0; i_input<inputs.rows(); i_input++)
270  {
271 
272  // Three main steps
273  // A: compute probability: prior * pdf of the multivariate Gaussian
274  // B: compute estimated mean of y: (mu_y + ( C_y_x * inv(C_x) * (input-mu_x) ) )
275  // C: weight the estimated mean with the probability
276 
277 
278  // A: compute probability: prior * pdf of the multivariate Gaussian
279  // Compute probalities that each Gaussian would generate this input in 4 steps
280  // A1. Compute the unnormalized pdf of the multi-variate Gaussian distribution
281  // A2. Normalize the unnormalized pdf (scale factor has been precomputed in the GMM)
282  // A3. Multiply the normalized pdf with the priors
283  // A4. Normalize the probabilities by dividing by their sum
284 
285  for (unsigned int i_gau=0; i_gau<gmm->getNumberOfGaussians(); i_gau++)
286  {
287  // A1. Compute the unnormalized pdf of the multi-variate Gaussian distribution
288  // formula: exp( -2 * (x-mu)^T * Sigma^-1 * (x-mu) )
289  // (we use cached variables and noalias to avoid dynamic allocation)
290  // (x-mu)
291  diff_prealloc_ = inputs.row(i_input).transpose() - gmm->means_x_[i_gau];
292  // Sigma^-1 * (x-mu)
293  covar_times_diff_prealloc_.noalias() = gmm->covars_x_inv_[i_gau]*diff_prealloc_;
294  // exp( -2 * (x-mu)^T * Sigma^-1 * (x-mu) )
295  probabilities_prealloc_[i_gau] = exp(-0.5*diff_prealloc_.dot(covar_times_diff_prealloc_));
296 
297  // A2. Normalize the unnormalized pdf (scale factor has been precomputed in the GMM)
298  // formula for scale factor: 1/sqrt( (2\pi)^N*|\Sigma| )
299  probabilities_prealloc_[i_gau] *= gmm->mvgd_scale_[i_gau];
300 
301  // A3. Multiply the normalized pdf with the priors
302  probabilities_prealloc_[i_gau] *= gmm->priors_[i_gau];
303 
304  }
305 
306  // A4. Normalize the probabilities by dividing by their sum
307  probabilities_prealloc_ /= probabilities_prealloc_.sum();
308 
309 
310  if (compute_means)
311  {
312  for (unsigned int i_gau=0; i_gau<gmm->getNumberOfGaussians(); i_gau++)
313  {
314 
315  // B: compute estimated mean of y: (mu_y + ( C_y_x * inv(C_x) * (input-mu_x) ) )
316  // We will compute it bit by bit (with preallocated matrices) to avoid dynamic allocations.
317 
318  // (input-mu_x)
319  diff_prealloc_ = inputs.row(i_input).transpose() - gmm->means_x_[i_gau];
320  // inv(C_x) * (input-mu_x)
321  covar_times_diff_prealloc_.noalias() = gmm->covars_x_inv_[i_gau]*diff_prealloc_;
322  // ( C_y_x * inv(C_x) * (input-mu_x) )
323  mean_output_prealloc_.noalias() = gmm->covars_y_x_[i_gau]*covar_times_diff_prealloc_;
324  // (mu_y + ( C_y_x * inv(C_x) * (input-mu_x) ) )
325  mean_output_prealloc_ += gmm->means_y_[i_gau];
326 
327  // C: weight the estimated mean with the probability
328  // probability * (mu_y + ( C_y_x * inv(C_x) * (input-mu_x) ) )
329  outputs.row(i_input) += probabilities_prealloc_[i_gau] * mean_output_prealloc_;
330  }
331  }
332 
333  if (compute_variance)
334  {
335  for (unsigned int i_gau=0; i_gau<gmm->getNumberOfGaussians(); i_gau++)
336  {
337  // Here comes the formula: h^2 * (C_y - C_y_x * inv(C_x) * C_y_x^T)
338 
339  // inv(C_x) * C_y_x^T
340  covar_input_times_output_.noalias() = gmm->covars_x_inv_[i_gau]*gmm->covars_y_x_[i_gau].transpose();
341  // - C_y_x * inv(C_x) * C_y_x^T
342  covar_output_prealloc_.noalias() = - gmm->covars_y_x_[i_gau] * covar_input_times_output_;
343  // (C_y - C_y_x * inv(C_x) * C_y_x^T)
344  covar_output_prealloc_ += gmm->covars_y_[i_gau];
345  // h^2 * (C_y - C_y_x * inv(C_x) * C_y_x^T)
346  variances.row(i_input) += probabilities_prealloc_[i_gau]*probabilities_prealloc_[i_gau] * ( covar_output_prealloc_ ).diagonal();
347 
348  // There are cases where we may get slightly negative variances due to numerical issues
349  // Avoid them here by setting negative variances to 0.
350  for (int i_output_dim=0; i_output_dim<gmm->getExpectedOutputDim(); i_output_dim++)
351  if (variances(i_input,i_output_dim)<0.0)
352  variances(i_input,i_output_dim) = 0.0;
353 
354 
355  }
356  }
357  }

527 void FunctionApproximatorGMR::firstDimSlicingInit(const MatrixXd& data, std::vector<VectorXd>& centers, std::vector<double>& priors,
528  std::vector<MatrixXd>& covars)
529 {
530 
531  VectorXd first_dim = data.col(0);
532 
533  VectorXi assign(data.rows());
534  assign.setZero();
535 
536  double min_val = first_dim.minCoeff();
537  double max_val = first_dim.maxCoeff();
538 
539  for (int i_first_dim = 0; i_first_dim < first_dim.size(); i_first_dim++)
540  {
541  unsigned int center = int((first_dim[i_first_dim]-min_val)/(max_val-min_val)*centers.size());
542  if (center==centers.size())
543  center--;
544  assign[i_first_dim] = center;
545  }
546 
547  // Init means
548  VectorXi nbPoints = VectorXi::Zero(centers.size());
549  for (size_t i_gau = 0; i_gau < centers.size(); i_gau++)
550  centers[i_gau].setZero();
551  for (int iData = 0; iData < data.rows(); iData++)
552  {
553  centers[assign[iData]] += data.row(iData).transpose();
554  nbPoints[assign[iData]]++;
555  }
556  for (size_t i_gau = 0; i_gau < centers.size(); i_gau++)
557  centers[i_gau] /= nbPoints[i_gau];
558 
559  // Init covars
560  for (size_t i_gau = 0; i_gau < centers.size(); i_gau++)
561  covars[i_gau].setZero();
562  for (int iData = 0; iData < data.rows(); iData++)
563  covars[assign[iData]] += (data.row(iData).transpose() - centers[assign[iData]]) * (data.row(iData).transpose() - centers[assign[iData]]).transpose();
564  for (size_t i_gau = 0; i_gau < centers.size(); i_gau++)
565  covars[i_gau] /= nbPoints[i_gau];
566 
567  // Be sure that covar is invertible
568  for (size_t i_gau = 0; i_gau < centers.size(); i_gau++)
569  covars[i_gau] += MatrixXd::Identity(covars[i_gau].rows(), covars[i_gau].cols()) * 1e-5;
570 
571  // Init priors
572  for (size_t i_gau = 0; i_gau < centers.size(); i_gau++)
573  priors[i_gau] = 1. / centers.size();

576 void FunctionApproximatorGMR::kMeansInit(const MatrixXd& data, std::vector<VectorXd>& centers, std::vector<double>& priors,
577  std::vector<MatrixXd>& covars, int n_max_iter)
578 {
579 
580  MatrixXd dataCentered = data.rowwise() - data.colwise().mean();
581  MatrixXd dataCov = dataCentered.transpose() * dataCentered / data.rows();
582  MatrixXd dataCovInverse = dataCov.inverse();
583 
584  std::vector<int> dataIndex;
585  for (int i = 0; i < data.rows(); i++)
586  dataIndex.push_back(i);
587  std::random_shuffle (dataIndex.begin(), dataIndex.end());
588 
589  for (size_t i_gau = 0; i_gau < centers.size(); i_gau++)
590  centers[i_gau] = data.row(dataIndex[i_gau]);
591 
592  VectorXi assign(data.rows());
593  assign.setZero();
594 
595  bool converged = false;
596  for (int iIter = 0; iIter < n_max_iter && !converged; iIter++)
597  {
598  //cout << " iIter=" << iIter << endl;
599 
600  // E step
601  converged = true;
602  for (int iData = 0; iData < data.rows(); iData++)
603  {
604  VectorXd v = (centers[assign[iData]] - data.row(iData).transpose());
605 
606  double minDist = v.transpose() * dataCovInverse * v;
607 
608  for (int i_gau = 0; i_gau < (int)centers.size(); i_gau++)
609  {
610  if (i_gau == assign[iData])
611  continue;
612 
613  v = (centers[i_gau] - data.row(iData).transpose());
614  double dist = v.transpose() * dataCovInverse * v;
615  if (dist < minDist)
616  {
617  converged = false;
618  minDist = dist;
619  assign[iData] = i_gau;
620  }
621  }
622  }
623 
624  // M step
625  VectorXi nbPoints = VectorXi::Zero(centers.size());
626  for (size_t i_gau = 0; i_gau < centers.size(); i_gau++)
627  centers[i_gau].setZero();
628  for (int iData = 0; iData < data.rows(); iData++)
629  {
630  centers[assign[iData]] += data.row(iData).transpose();
631  nbPoints[assign[iData]]++;
632  }
633  for (size_t i_gau = 0; i_gau < centers.size(); i_gau++)
634  centers[i_gau] /= nbPoints[i_gau];
635  }
636 
637  // Init covars
638  VectorXi nbPoints = VectorXi::Zero(centers.size());
639  for (size_t i_gau = 0; i_gau < centers.size(); i_gau++)
640  covars[i_gau].setZero();
641  for (int iData = 0; iData < data.rows(); iData++)
642  {
643  covars[assign[iData]] += (data.row(iData).transpose() - centers[assign[iData]]) * (data.row(iData).transpose() - centers[assign[iData]]).transpose();
644  nbPoints[assign[iData]]++;
645  }
646  for (size_t i_gau = 0; i_gau < centers.size(); i_gau++)
647  covars[i_gau] /= nbPoints[i_gau];
648 
649  // Be sure that covar is invertible
650  for (size_t i_gau = 0; i_gau < centers.size(); i_gau++)
651  covars[i_gau] += MatrixXd::Identity(covars[i_gau].rows(), covars[i_gau].cols()) * 1e-5f;
652 
653  // Init priors
654  for (size_t i_gau = 0; i_gau < centers.size(); i_gau++)
655  priors[i_gau] = 1. / centers.size();
656 }
