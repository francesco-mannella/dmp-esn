71 void ExponentialSystem::analyticalSolution(const VectorXd& ts, MatrixXd& xs, MatrixXd& xds) const
72 {
73  int T = ts.size();
74  assert(T>0);
75 
76  // Usually, we expect xs and xds to be of size T X dim(), so we resize to that. However, if the
77  // input matrices were of size dim() X T, we return the matrices of that size by doing a
78  // transposeInPlace at the end. That way, the user can also request dim() X T sized matrices.
79  bool caller_expects_transposed = (xs.rows()==dim() && xs.cols()==T);
80 
81  // Prepare output arguments to be of right size (Eigen does nothing if already the right size)
82  xs.resize(T,dim());
83  xds.resize(T,dim());
84 
85  VectorXd val_range = initial_state() - attractor_state();
86 
87  VectorXd exp_term = -alpha_*ts/tau();
88  exp_term = exp_term.array().exp().transpose();
89  VectorXd pos_scale = exp_term;
90  VectorXd vel_scale = -(alpha_/tau()) * exp_term;
91 
92  xs = val_range.transpose().replicate(T,1).array() * pos_scale.replicate(1,dim()).array();
93  xs += attractor_state().transpose().replicate(T,1);
94  xds = val_range.transpose().replicate(T,1).array() * vel_scale.replicate(1,dim()).array();
95 
96  if (caller_expects_transposed)
97  {
98  xs.transposeInPlace();
99  xds.transposeInPlace();
100  }
101 }
