103 void MetaParametersRBFN::getCentersAndWidths(const VectorXd& min, const VectorXd& max, Eigen::MatrixXd& centers, Eigen::MatrixXd& widths) const
104 {
105  int n_dims = getExpectedInputDim();
106  assert(min.size()==n_dims);
107  assert(max.size()==n_dims);
108 
109  vector<VectorXd> centers_per_dim_local(n_dims);
110  if (!centers_per_dim_.empty())
111  {
112  centers_per_dim_local = centers_per_dim_;
113  }
114  else
115  {
116  // Centers are not know yet, compute them based on min and max
117  for (int i_dim=0; i_dim<n_dims; i_dim++)
118  centers_per_dim_local[i_dim] = VectorXd::LinSpaced(n_bfs_per_dim_[i_dim],min[i_dim],max[i_dim]);
119  }
120 
121  // Determine the widths from the centers (separately for each dimension)
122  vector<VectorXd> widths_per_dim_local(n_dims);
123  for (int i_dim=0; i_dim<n_dims; i_dim++)
124  {
125  VectorXd cur_centers = centers_per_dim_local[i_dim]; // Abbreviation for convenience
126  int n_centers = cur_centers.size();
127  VectorXd cur_widths(n_centers);
128 
129  if (n_centers==1)
130  {
131  cur_widths[0] = 1.0;
132  }
133  else
134  {
135  // Consider two neighbouring basis functions, exp(-0.5(x-c0)^2/w^2) and exp(-0.5(x-c1)^2/w^2)
136  // Assuming the widths are the same for both, they are certain to intersect at x = 0.5(c0+c1)
137  // And we want the activation at x to be 'intersection'. So
138  // y = exp(-0.5(x-c0)^2/w^2)
139  // intersection = exp(-0.5((0.5(c0+c1))-c0)^2/w^2)
140  // intersection = exp(-0.5((0.5*c1-0.5*c0)^2/w^2))
141  // intersection = exp(-0.5((0.5*(c1-c0))^2/w^2))
142  // intersection = exp(-0.5(0.25*(c1-c0)^2/w^2))
143  // intersection = exp(-0.125((c1-c0)^2/w^2))
144  // w = sqrt((c1-c0)^2/-8*ln(intersection))
145  for (int cc=0; cc<n_centers-1; cc++)
146  {
147  double w = sqrt(pow(cur_centers[cc+1]-cur_centers[cc],2)/(-8*log(intersection_height_)));
148  cur_widths[cc] = w;
149  }
150  cur_widths[n_centers-1] = cur_widths[n_centers-2];
151  }
152  widths_per_dim_local[i_dim] = cur_widths;
153  }
154 
155  VectorXd digit_max(n_dims);
156  int n_centers = 1;
157  for (int i_dim=0; i_dim<n_dims; i_dim++)
158  {
159  n_centers = n_centers*centers_per_dim_local[i_dim].size();
160  digit_max[i_dim] = centers_per_dim_local[i_dim].size();
161  }
162  VectorXi digit = VectorXi::Zero(n_dims);
163 
164  centers.resize(n_centers,n_dims);
165  widths.resize(n_centers,n_dims);
166  int i_center=0;
167 
168  while (digit[0]<digit_max(0))
169  {
170  for (int i_dim=0; i_dim<n_dims; i_dim++)
171  {
172  centers(i_center,i_dim) = centers_per_dim_local[i_dim][digit[i_dim]];
173  widths(i_center,i_dim) = widths_per_dim_local[i_dim][digit[i_dim]];
174  }
175  i_center++;
176 
177  // Increment last digit by one
178  digit[n_dims-1]++;
179  for (int i_dim=n_dims-1; i_dim>0; i_dim--)
180  {
181  if (digit[i_dim]>=digit_max[i_dim])
182  {
183  digit[i_dim] = 0;
184  digit[i_dim-1]++;
185  }
186  }
187  }
