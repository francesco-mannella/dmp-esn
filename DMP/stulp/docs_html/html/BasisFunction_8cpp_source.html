110 void Gaussian::activations(const Eigen::MatrixXd& centers, const Eigen::MatrixXd& widths, const Eigen::MatrixXd& inputs, Eigen::MatrixXd& kernel_activations, bool normalized_basis_functions, bool asymmetric_kernels)
111 {
112 
113  // Check and set sizes
114  // centers = n_basis_functions x n_dim
115  // widths = n_basis_functions x n_dim
116  // inputs = n_samples x n_dim
117  // activations = n_samples x n_basis_functions
118  int n_basis_functions = centers.rows();
119  int n_samples = inputs.rows();
120  int n_dims = centers.cols();
121  assert( (n_basis_functions==widths.rows()) & (n_dims==widths.cols()) );
122  assert( (n_samples==inputs.rows() ) & (n_dims==inputs.cols()) );
123  kernel_activations.resize(n_samples,n_basis_functions);
124 
125  if (normalized_basis_functions && n_basis_functions==1)
126  {
127  // Locally Weighted Regression with only one basis function is pretty odd.
128  // Essentially, you are taking the "Locally Weighted" part out of the regression, and it becomes
129  // standard least squares
130  // Anyhow, for those that still want to "abuse" LWR as R (i.e. without LW), we explicitly
131  // set the normalized kernels to 1 here, to avoid numerical issues in the normalization below.
132  // (normalizing a Gaussian basis function with itself leads to 1 everywhere).
133  kernel_activations.fill(1.0);
134  return;
135  }
136 
137  double c,w,x;
138  for (int bb=0; bb<n_basis_functions; bb++)
139  {
140 
141  // Here, we compute the values of a (unnormalized) multi-variate Gaussian:
142  // activation = exp(-0.5*(x-mu)*Sigma^-1*(x-mu))
143  // Because Sigma is diagonal in our case, this simplifies to
144  // activation = exp(\sum_d=1^D [-0.5*(x_d-mu_d)^2/Sigma_(d,d)])
145  // = \prod_d=1^D exp(-0.5*(x_d-mu_d)^2/Sigma_(d,d))
146  // This last product is what we compute below incrementally
147 
148  kernel_activations.col(bb).fill(1.0);
149  for (int i_dim=0; i_dim<n_dims; i_dim++)
150  {
151  c = centers(bb,i_dim);
152  for (int i_s=0; i_s<n_samples; i_s++)
153  {
154  x = inputs(i_s,i_dim);
155  w = widths(bb,i_dim);
156 
157  if (asymmetric_kernels && x<c && bb>0)
158  // Get the width of the previous basis function
159  // This is the part that makes it assymetric
160  w = widths(bb-1,i_dim);
161 
162  kernel_activations(i_s,bb) *= exp(-0.5*pow(x-c,2)/(w*w));
163  }
164  }
165  }
166 
167  if (normalized_basis_functions)
168  {
169  // Compute sum for each row (each value in input_vector)
170  MatrixXd sum_kernel_activations = kernel_activations.rowwise().sum(); // n_samples x 1
171

177  kernel_activations = kernel_activations.array()/sum_kernel_activations.replicate(1,n_basis_functions).array();
