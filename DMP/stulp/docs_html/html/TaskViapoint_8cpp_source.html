65 void TaskViapoint::evaluate(const MatrixXd& cost_vars, const MatrixXd& task_parameters, VectorXd& costs) const
66 {
67  int n_samples = cost_vars.rows();
68  int n_dims = viapoint_.size();
69  int n_cost_vars = 4*n_dims + 1; // y_1..y_D yd_1..yd_D ydd_1..ydd_D t forcing_term_1..forcing_term_D
70  int n_time_steps = cost_vars.cols()/n_cost_vars;
71  // cost_vars = n_samples x (n_time_steps*n_cost_vars)
72 
73  //cout << " n_samples=" << n_samples << endl;
74  //cout << " n_dims=" << n_dims << endl;
75  //cout << " n_cost_vars=" << n_cost_vars << endl;
76  //cout << " cost_vars.cols()=" << cost_vars.cols() << endl;
77  //cout << " n_time_steps=" << n_time_steps << endl;
78 
79  costs.resize(n_samples);
80 
81  // y_1..y_D yd_1..yd_D ydd_1..ydd_D t=0
82  // y_1..y_D yd_1..yd_D ydd_1..ydd_D t=1
83  // | | | | | | |
84  // y_1..y_D yd_1..yd_D ydd_1..ydd_D t=T
85  MatrixXd rollout; //(n_time_steps,n_cost_vars);
86  MatrixXd my_row(1,n_time_steps*n_cost_vars);
87 
88  for (int k=0; k<n_samples; k++)
89  {
90  my_row = cost_vars.row(k);
91  rollout = (Map<MatrixXd>(my_row.data(),n_cost_vars,n_time_steps)).transpose();
92 
93  // rollout is of size n_time_steps x n_cost_vars
94  VectorXd ts = rollout.col(3 * n_dims);
95 
96  double dist_to_viapoint = 0.0;
97  if (viapoint_weight_!=0.0)
98  {
99  if (viapoint_time_ == TIME_AT_MINIMUM_DIST)
100  {
101  // Don't compute the distance at some time, but rather get the minimum distance
102  const MatrixXd y = rollout.block(0, 0, rollout.rows(), n_dims);
103  dist_to_viapoint = (y.rowwise() - viapoint_.transpose()).rowwise().squaredNorm().minCoeff();
104  }
105  else
106  {
107  // Compute the minimum distance at a specific time
108 
109  // Get the time_step at which viapoint_time_step approx ts[time_step]
110  int viapoint_time_step = 0;
111  while (viapoint_time_step < ts.size() && ts[viapoint_time_step] < viapoint_time_)
112  viapoint_time_step++;
113 
114  assert(viapoint_time_step < ts.size());
115 
116  VectorXd y_via = rollout.row(viapoint_time_step).segment(0,n_dims);
117  dist_to_viapoint = sqrt((y_via-viapoint_).array().pow(2).sum());
118  }
119 
120  if (viapoint_radius_>0.0)
121  {
122  // The viapoint_radius defines a radius within which the cost is always 0
123  dist_to_viapoint -= viapoint_radius_;
124  if (dist_to_viapoint<0.0)
125  dist_to_viapoint = 0.0;
126  }
127  }
128 
129  double sum_ydd = 0.0;
130  if (acceleration_weight_!=0.0)
131  {
132  MatrixXd ydd = rollout.block(0,2*n_dims,n_time_steps,n_dims);
133  // ydd = n_time_steps x n_dims
134  sum_ydd = ydd.array().pow(2).sum();
135  }
136 
137  double delay_cost = 0.0;
138  if (goal_weight_!=0.0)
139  {
140  int goal_time_step = 0;
141  while (goal_time_step < ts.size() && ts[goal_time_step] < goal_time_)
142  goal_time_step++;
143 
144  const MatrixXd y_after_goal = rollout.block(goal_time_step, 0,
145  rollout.rows() - goal_time_step, n_dims);
146 
147  delay_cost = (y_after_goal.rowwise() - goal_.transpose()).rowwise().squaredNorm().sum();
148  }
149 
150  costs[k] =
151  viapoint_weight_*dist_to_viapoint +
152  acceleration_weight_*sum_ydd/n_time_steps +
153  goal_weight_*delay_cost;
154  }
155 }
