135 Trajectory Trajectory::generateMinJerkTrajectory(const VectorXd& ts, const VectorXd& y_from, const VectorXd& y_to)
136 {
137  int n_time_steps = ts.size();
138  int n_dims = y_from.size();
139 
140  MatrixXd ys(n_time_steps,n_dims), yds(n_time_steps,n_dims), ydds(n_time_steps,n_dims);
141 
142  double D = ts[n_time_steps-1];
143  ArrayXd tss = (ts/D).array();
144 
145 
146  ArrayXd A = y_to.array()-y_from.array();
147 
148  for (int i_dim=0; i_dim<n_dims; i_dim++)
149  {
150 
151  // http://noisyaccumulation.blogspot.fr/2012/02/how-to-decompose-2d-trajectory-data.html
152 
153  ys.col(i_dim) = y_from[i_dim] + A[i_dim]*( 6*tss.pow(5) -15*tss.pow(4) +10*tss.pow(3));
154 
155  yds.col(i_dim) = (A[i_dim]/D)*( 30*tss.pow(4) -60*tss.pow(3) +30*tss.pow(2));
156 
157  ydds.col(i_dim) = (A[i_dim]/(D*D))*(120*tss.pow(3) -180*tss.pow(2) +60*tss );
158  }
159 
160  return Trajectory(ts,ys,yds,ydds);

165 Trajectory Trajectory::generatePolynomialTrajectory(const VectorXd& ts, const VectorXd& y_from, const VectorXd& yd_from, const VectorXd& ydd_from,
166  const VectorXd& y_to, const VectorXd& yd_to, const VectorXd& ydd_to)
167 {
168  VectorXd a0 = y_from;
169  VectorXd a1 = yd_from;
170  VectorXd a2 = ydd_from / 2;
171 
172  VectorXd a3 = -10 * y_from - 6 * yd_from - 2.5 * ydd_from + 10 * y_to - 4 * yd_to + 0.5 * ydd_to;
173  VectorXd a4 = 15 * y_from + 8 * yd_from + 2 * ydd_from - 15 * y_to + 7 * yd_to - ydd_to;
174  VectorXd a5 = -6 * y_from - 3 * yd_from - 0.5 * ydd_from + 6 * y_to - 3 * yd_to + 0.5 * ydd_to;
175 
176  int n_time_steps = ts.size();
177  int n_dims = y_from.size();
178 
179  MatrixXd ys(n_time_steps,n_dims), yds(n_time_steps,n_dims), ydds(n_time_steps,n_dims);
180 
181  for (int i = 0; i < ts.size(); i++)
182  {
183  double t = (ts[i] - ts[0]) / (ts[n_time_steps - 1] - ts[0]);
184  ys.row(i) = a0 + a1 * t + a2 * pow(t, 2) + a3 * pow(t, 3) + a4 * pow(t, 4) + a5 * pow(t, 5);
185  yds.row(i) = a1 + 2 * a2 * t + 3 * a3 * pow(t, 2) + 4 * a4 * pow(t, 3) + 5 * a5 * pow(t, 4);
186  ydds.row(i) = 2 * a2 + 6 * a3 * t + 12 * a4 * pow(t, 2) + 20 * a5 * pow(t, 3);
187  }
188 
189  yds /= (ts[n_time_steps - 1] - ts[0]);
190  ydds /= pow(ts[n_time_steps - 1] - ts[0], 2);
191 
192  return Trajectory(ts, ys, yds, ydds);
193 }
