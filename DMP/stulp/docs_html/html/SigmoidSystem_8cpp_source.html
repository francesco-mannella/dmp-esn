81 VectorXd SigmoidSystem::computeKs(const VectorXd& N_0s, double r, double inflection_point_time_time)
82 {
83  // The idea here is that the initial state (called N_0s above), max_rate (r above) and the
84  // inflection_point_time are set by the user.
85  // The only parameter that we have left to tune is the "carrying capacity" K.
86  // http://en.wikipedia.org/wiki/Logistic_function#In_ecology:_modeling_population_growth
87  // In the below, we set K so that the initial state is N_0s for the given r and tau
88 
89  // Known
90  // N(t) = K / ( 1 + (K/N_0 - 1)*exp(-r*t))
91  // N(t_inf) = K / 2
92  // Plug into each other and solve for K
93  // K / ( 1 + (K/N_0 - 1)*exp(-r*t_infl)) = K/2
94  // (K/N_0 - 1)*exp(-r*t_infl) = 1
95  // (K/N_0 - 1) = 1/exp(-r*t_infl)
96  // K = N_0*(1+(1/exp(-r*t_infl)))
97  VectorXd Ks = N_0s;
98  for (int dd=0; dd<Ks.size(); dd++)
99  Ks[dd] = N_0s[dd]*(1+(1/exp(-r*inflection_point_time_time)));
100 
101 
102  // If Ks is too close to N_0===initial_state, then the differential equation will always return 0
103  // See differentialEquation below
104  // xd = max_rate_*x*(1-(x/Ks_));
105  // For initial_state this is
106  // xd = max_rate_*initial_state*(1-(initial_state/Ks_));
107  // If initial_state is very close/equal to Ks we get
108  // xd = max_rate_*Ks*(1-(Ks/Ks_));
109  // xd = max_rate_*Ks*(1-1);
110  // xd = max_rate_*Ks*0;
111  // xd = 0;
112  // And integration fails, especially for Euler integration.
113  // So we now give a warning if this is likely to happen.
114  VectorXd div = N_0s.array()/Ks.array()-1;
115  //cout << setprecision(20) << div << endl;
116  if ((div.array().abs() < 10e-9).any()) // 10e-9 determined empirically
117  {
118 
119  ostringstream message;
120  message << "In function SigmoidSystem::computeKs(), Ks is too close to N_0s."
121  "This may lead to errors during numerical integration. Recommended "
122  "solution: choose a lower magnitude for the maximum rate of change "
123  "(currently it is "<<r <<")";
124 
125  cerr << endl << __FILE__ << ":" << __LINE__ << ":";
126  cerr << message.str() << endl;
127  }
128 
129  return Ks;
130 }

137 void SigmoidSystem::analyticalSolution(const VectorXd& ts, MatrixXd& xs, MatrixXd& xds) const
138 {
139  int T = ts.size();
140  assert(T>0);
141 
142  // Usually, we expect xs and xds to be of size T X dim(), so we resize to that. However, if the
143  // input matrices were of size dim() X T, we return the matrices of that size by doing a
144  // transposeInPlace at the end. That way, the user can also request dim() X T sized matrices.
145  bool caller_expects_transposed = (xs.rows()==dim() && xs.cols()==T);
146 
147  // Prepare output arguments to be of right size (Eigen does nothing if already the right size)
148  xs.resize(T,dim());
149  xds.resize(T,dim());
150 
151  // Auxillary variables to improve legibility
152  double r = max_rate_;
153  VectorXd exp_rt = (-r*ts).array().exp();
154 
155  VectorXd y_init = initial_state();
156 
157  for (int dd=0; dd<dim(); dd++)
158  {
159  // Auxillary variables to improve legibility
160  double K = Ks_[dd];
161  double b = (K/y_init[dd])-1;
162 
163  xs.block(0,dd,T,1) = K/(1+b*exp_rt.array());
164  xds.block(0,dd,T,1) = K*r*b*((1 + b*exp_rt.array()).square().inverse().array()) * exp_rt.array();
165  }
166 
167  if (caller_expects_transposed)
168  {
169  xs.transposeInPlace();
170  xds.transposeInPlace();
171  }
172 }
