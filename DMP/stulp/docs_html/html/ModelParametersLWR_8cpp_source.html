126 void ModelParametersLWR::set_lines_pivot_at_max_activation(bool lines_pivot_at_max_activation)
127 {
128  // If no change, just return
129  if (lines_pivot_at_max_activation_ == lines_pivot_at_max_activation)
130  return;
131 
132  //cout << "________________" << endl;
133  //cout << centers_.transpose() << endl;
134  //cout << slopes_.transpose() << endl;
135  //cout << offsets_.transpose() << endl;
136  //cout << "centers_ = " << centers_.rows() << "X" << centers_.cols() << endl;
137  //cout << "slopes_ = " << slopes_.rows() << "X" << slopes_.cols() << endl;
138  //cout << "offsets_ = " << offsets_.rows() << "X" << offsets_.cols() << endl;
139 
140  // If you pivot lines around the point when the basis function has maximum activation (i.e.
141  // at the center of the Gaussian), you must compute the new offset corresponding to this
142  // slope, and vice versa
143  int n_lines = centers_.rows();
144  VectorXd ac(n_lines); // slopes*centers
145  for (int i_line=0; i_line<n_lines; i_line++)
146  {
147  ac[i_line] = slopes_.row(i_line) * centers_.row(i_line).transpose();
148  }
149 
150  if (lines_pivot_at_max_activation)
151  {
152  // Representation was "y = ax + b", now it will be "y = a(x-c) + b^new"
153  // Since "y = ax + b" can be rewritten as "y = a(x-c) + (b+ac)", we know that "b^new = (ac+b)"
154  offsets_ = offsets_ + ac;
155  }
156  else
157  {
158  // Representation was "y = a(x-c) + b", now it will be "y = ax + b^new"
159  // Since "y = a(x-c) + b" can be rewritten as "y = ax + (b-ac)", we know that "b^new = (b-ac)"
160  offsets_ = offsets_ - ac;
161  }
162  // Remark, the above could have been done as a one-liner, but I prefer the more legible version.
163 
164  //cout << offsets_.transpose() << endl;
165  //cout << "offsets_ = " << offsets_.rows() << "X" << offsets_.cols() << endl;
166 
167  lines_pivot_at_max_activation_ = lines_pivot_at_max_activation;
168 }

179 void ModelParametersLWR::getLines(const MatrixXd& inputs, MatrixXd& lines) const
180 {
181  int n_time_steps = inputs.rows();
182 
183  //cout << "centers_ = " << centers_.rows() << "X" << centers_.cols() << endl;
184  //cout << "slopes_ = " << slopes_.rows() << "X" << slopes_.cols() << endl;

186  //cout << "inputs = " << inputs.rows() << "X" << inputs.cols() << endl;
187 
188  // Compute values along lines for each time step
189  // Line representation is "y = ax + b"
190  lines = inputs*slopes_.transpose() + offsets_.transpose().replicate(n_time_steps,1);
191 
192  if (lines_pivot_at_max_activation_)
193  {
194  // Line representation is "y = a(x-c) + b", which is "y = ax - ac + b"
195  // Therefore, we still have to subtract "ac"
196  int n_lines = centers_.rows();
197  VectorXd ac(n_lines); // slopes*centers = ac
198  for (int i_line=0; i_line<n_lines; i_line++)
199  ac[i_line] = slopes_.row(i_line) * centers_.row(i_line).transpose();
200  //cout << "ac = " << ac.rows() << "X" << ac.cols() << endl;
201  lines = lines - ac.transpose().replicate(n_time_steps,1);
202  }
203  //cout << "lines = " << lines.rows() << "X" << lines.cols() << endl;
