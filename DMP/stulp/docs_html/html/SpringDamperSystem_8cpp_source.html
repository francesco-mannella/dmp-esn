65 void SpringDamperSystem::differentialEquation(const Eigen::VectorXd& x, Eigen::Ref<Eigen::VectorXd> xd) const
66 {
67  // Spring-damper system was originally 2nd order, i.e. with [x xd xdd]
68  // After rewriting it as a 1st order system it becomes [y z yd zd], with yd = z;
69 
70  // Get 'y' and 'z' parts of the state in 'x'
71  int dim2 = dim()/2;
72  VectorXd y = x.segment(0,dim2);
73  VectorXd z = x.segment(dim2,dim2);
74  VectorXd y_attr = attractor_state().segment(0,dim2);
75 
76  // Compute yd and zd
77  // See http://en.wikipedia.org/wiki/Damped_spring-mass_system#Example:mass_.E2.80.93spring.E2.80.93damper
78  // and equation 2.1 of http://www-clmc.usc.edu/publications/I/ijspeert-NC2013.pdf
79 
80  VectorXd yd = z/tau();
81 
82  VectorXd zd = (-spring_constant_*(y-y_attr) - damping_coefficient_*z)/(mass_*tau());
83 
84  xd.segment(0,dim()/2) = yd;
85  xd.segment(dim()/2,dim()/2) = zd;
86 
87 }

89 void SpringDamperSystem::analyticalSolution(const VectorXd& ts, MatrixXd& xs, MatrixXd& xds) const
90 {
91  int T = ts.size();
92  assert(T>0);
93 
94  // Usually, we expect xs and xds to be of size T X dim(), so we resize to that. However, if the
95  // input matrices were of size dim() X T, we return the matrices of that size by doing a
96  // transposeInPlace at the end. That way, the user can also request dim() X T sized matrices.
97  bool caller_expects_transposed = (xs.rows()==dim() && xs.cols()==T);
98 
99  // Prepare output arguments to be of right size (Eigen does nothing if already the right size)
100  xs.resize(T,dim());
101  xds.resize(T,dim());
102 
103  VectorXd y_init = initial_state();
104  VectorXd y_attr = attractor_state();
105 
106  // Closed form solution to 2nd order canonical system
107  // This system behaves like a critically damped spring-damper system
108  // http://en.wikipedia.org/wiki/Damped_spring-mass_system
109  double omega_0 = sqrt(spring_constant_/mass_)/tau(); // natural frequency
110  double zeta = damping_coefficient_/(2*sqrt(mass_*spring_constant_)); // damping ratio
111  if (zeta!=1.0)
112  cout << "WARNING: Spring-damper system is not critically damped, zeta=" << zeta << endl;
113 
114  // Example
115  // _______________
116  // dim = 4
117  // _______
118  // dim2= 2
119  // [y_1 y_2 z_1 z_2 yd_1 yd_2 zd_1 zd_2]
120 
121  int dim2 = dim()/2;
122 
123  // The loop is slower, but more legible than fudging around with Eigen::replicate().
124  for (int i_dim=0; i_dim<dim2; i_dim++)
125  {
126  double y0 = y_init[i_dim] - y_attr[i_dim];
127  double yd0;
128  if (y_init.size()>=dim())
129  {
130  // Initial velocities also given
131  yd0 = y_init[dim2 + i_dim];
132  }
133  else
134  {
135  // Initial velocities not given: set to zero
136  yd0 = 0.0;
137  }
138 
139  double A = y0;
140  double B = yd0 + omega_0*y0;
141 
142  // Closed form solutions
143  // See http://en.wikipedia.org/wiki/Damped_spring-mass_system
144  VectorXd exp_term = -omega_0*ts;
145  exp_term = exp_term.array().exp();
146 
147  int Y = 0*dim2 + i_dim;
148  int Z = 1*dim2 + i_dim;
149 
150  VectorXd ABts = A + B*ts.array();
151 
152  // Closed form solutions
153  // See http://en.wikipedia.org/wiki/Damped_spring-mass_system
154  // If you find this illegible, I recommend to look at the Matlab code instead...
155  xs.col(Y) = y_attr(i_dim) + (( ABts.array()))*exp_term.array();
156 
157  // Derivative of the above (use product rule: (f*g)' = f'*g + f*g'
158  xds.col(Y) = ((B - omega_0* ABts.array()))*exp_term.array();
159 
160  // Derivative of the above (again use product rule: (f*g)' = f'*g + f*g'
161  VectorXd ydds = (-omega_0*(2*B - omega_0* ABts.array()))*exp_term.array();
162 
163  // This is how to compute the 'z' terms from the 'y' terms
164  xs.col(Z) = xds.col(Y)*tau();
165  xds.col(Z) = ydds*tau();
166  }
167 
168  if (caller_expects_transposed)
169  {
170  xs.transposeInPlace();
171  xds.transposeInPlace();
172  }
173 }
